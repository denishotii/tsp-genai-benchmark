```python
import numpy as np

# Step-by-step reasoning:
#
# 1. Problem restatement and key algorithm components:
#    We need to build a closed TSP tour using the nearest-neighbor greedy
#    heuristic. Starting from a fixed city, here city 0, we repeatedly choose
#    the closest city that has not yet been visited. Once all cities have been
#    visited, we return from the final city back to the starting city.
#
# 2. Data structures and main loop:
#    - `tour`: a Python list storing the order in which cities are visited.
#    - `visited`: a Boolean NumPy array where visited[i] indicates whether
#      city i has already been included in the tour.
#    - `current`: the current city.
#    - `length`: running total of the closed tour length.
#    In each iteration, we look at the current city's distance row, mask out
#    already visited cities by setting their distances to infinity, then choose
#    the minimum-distance unvisited city.
#
# 3. Edge cases:
#    - n == 0: no cities, so return an empty tour with length 0.0.
#    - n == 1: the tour is [0], and returning to itself adds distance 0.
#    - Ties: NumPy's argmin returns the first minimum index, so ties are broken
#      deterministically by choosing the smallest city index.


def solve(dist_matrix):
    dist_matrix = np.asarray(dist_matrix, dtype=float)
    n = dist_matrix.shape[0]

    if n == 0:
        return [], 0.0

    start = 0
    current = start

    tour = [start]
    visited = np.zeros(n, dtype=bool)
    visited[start] = True

    length = 0.0

    for _ in range(n - 1):
        distances = dist_matrix[current].copy()
        distances[visited] = np.inf

        next_city = int(np.argmin(distances))

        length += float(dist_matrix[current, next_city])
        tour.append(next_city)
        visited[next_city] = True
        current = next_city

    length += float(dist_matrix[current, start])

    return tour, float(length)
```
