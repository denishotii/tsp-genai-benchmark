import math
import numpy as np


def solve(dist_matrix):
    dist_matrix = np.asarray(dist_matrix, dtype=float)
    n = dist_matrix.shape[0]

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0
    if n == 2:
        return [0, 1], float(dist_matrix[0, 1] + dist_matrix[1, 0])

    rng = np.random.default_rng()

    def tour_length(tour):
        return float(
            sum(dist_matrix[tour[i], tour[(i + 1) % n]] for i in range(n))
        )

    def two_opt_delta(tour, i, j):
        a = tour[i - 1]
        b = tour[i]
        c = tour[j]
        d = tour[(j + 1) % n]
        return (
            dist_matrix[a, c]
            + dist_matrix[b, d]
            - dist_matrix[a, b]
            - dist_matrix[c, d]
        )

    # Nearest-neighbor initial tour
    tour = np.empty(n, dtype=int)
    unvisited = np.ones(n, dtype=bool)

    current = 0
    tour[0] = current
    unvisited[current] = False

    for k in range(1, n):
        dists = dist_matrix[current].copy()
        dists[~unvisited] = np.inf
        current = int(np.argmin(dists))
        tour[k] = current
        unvisited[current] = False

    current_length = tour_length(tour)
    best_tour = tour.copy()
    best_length = current_length

    # Estimate an initial temperature from sampled worsening 2-opt moves
    positive_deltas = []
    samples = min(2000, n * (n - 1) // 2)

    for _ in range(samples):
        i = int(rng.integers(0, n - 1))
        j = int(rng.integers(i + 1, n))

        if i == 0 and j == n - 1:
            continue

        delta = two_opt_delta(tour, i, j)
        if delta > 0:
            positive_deltas.append(delta)

    if positive_deltas:
        avg_delta = float(np.mean(positive_deltas))
        temperature = avg_delta / (-math.log(0.8))
    else:
        temperature = max(current_length / n, 1.0)

    final_temperature = max(temperature * 1e-4, 1e-12)

    # Geometric cooling schedule
    max_moves = int(min(2_000_000, max(50_000, 2000 * n)))
    moves_per_temperature = max(100, min(10 * n, 5000))
    temperature_steps = max(50, max_moves // moves_per_temperature)

    alpha = (final_temperature / temperature) ** (1.0 / temperature_steps)

    moves_done = 0

    while temperature > final_temperature and moves_done < max_moves:
        for _ in range(moves_per_temperature):
            if moves_done >= max_moves:
                break

            i = int(rng.integers(0, n - 1))
            j = int(rng.integers(i + 1, n))

            # Reversing the whole cyclic tour changes nothing
            if i == 0 and j == n - 1:
                continue

            delta = two_opt_delta(tour, i, j)

            if delta <= 0 or rng.random() < math.exp(-delta / temperature):
                tour[i:j + 1] = tour[i:j + 1][::-1].copy()
                current_length += delta

                if current_length < best_length:
                    best_length = current_length
                    best_tour = tour.copy()

            moves_done += 1

        temperature *= alpha

    # Final deterministic 2-opt improvement pass from the best SA solution
    tour = best_tour.copy()
    current_length = best_length

    improved = True
    passes = 0
    max_passes = 30 if n <= 400 else 5

    while improved and passes < max_passes:
        improved = False
        passes += 1

        for i in range(n - 1):
            if improved:
                break

            for j in range(i + 1, n):
                if i == 0 and j == n - 1:
                    continue

                delta = two_opt_delta(tour, i, j)

                if delta < -1e-12:
                    tour[i:j + 1] = tour[i:j + 1][::-1].copy()
                    current_length += delta
                    improved = True
                    break

    result_tour = tour.tolist()
    result_length = tour_length(result_tour)

    return result_tour, float(result_length)
