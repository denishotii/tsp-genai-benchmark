import numpy as np


def solve(dist_matrix):
    dist = np.asarray(dist_matrix, dtype=float)
    n = dist.shape[0]

    if dist.ndim != 2 or dist.shape[0] != dist.shape[1]:
        raise ValueError("dist_matrix must be a square NumPy array")

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0
    if n == 2:
        tour = [0, 1]
        length = float(dist[0, 1] + dist[1, 0])
        return tour, length

    rng = np.random.default_rng()

    def tour_length(tour):
        return float(dist[tour, np.roll(tour, -1)].sum())

    def population_lengths(pop):
        return dist[pop, np.roll(pop, -1, axis=1)].sum(axis=1)

    def nearest_neighbor(start):
        tour = np.empty(n, dtype=np.int64)
        unvisited = np.ones(n, dtype=bool)

        current = start
        for i in range(n):
            tour[i] = current
            unvisited[current] = False

            if i == n - 1:
                break

            candidates = np.flatnonzero(unvisited)
            current = candidates[np.argmin(dist[current, candidates])]

        return tour

    def tournament_select(pop, lengths, tournament_size):
        competitors = rng.integers(0, len(pop), size=tournament_size)
        winner = competitors[np.argmin(lengths[competitors])]
        return pop[winner]

    def order_crossover(parent1, parent2):
        child = np.empty(n, dtype=np.int64)
        child.fill(-1)

        a, b = sorted(rng.choice(n, size=2, replace=False))
        b += 1

        child[a:b] = parent1[a:b]

        used = np.zeros(n, dtype=bool)
        used[parent1[a:b]] = True

        pos = b % n
        for city in np.concatenate((parent2[b:], parent2[:b])):
            if not used[city]:
                child[pos] = city
                used[city] = True
                pos = (pos + 1) % n

        return child

    def swap_mutation(tour, mutation_rate):
        if rng.random() < mutation_rate:
            i, j = rng.choice(n, size=2, replace=False)
            tour[i], tour[j] = tour[j], tour[i]

    pop_size = int(np.clip(8 * n, 80, 300))
    generations = int(np.clip(1200 - n, 250, 1000))
    elite_count = max(1, pop_size // 20)
    tournament_size = 4
    mutation_rate = 0.20

    population = np.empty((pop_size, n), dtype=np.int64)

    seed_count = min(pop_size // 5, n, 20)
    starts = rng.choice(n, size=seed_count, replace=False)

    for i, start in enumerate(starts):
        population[i] = nearest_neighbor(int(start))

    for i in range(seed_count, pop_size):
        population[i] = rng.permutation(n)

    lengths = population_lengths(population)

    best_idx = int(np.argmin(lengths))
    best_tour = population[best_idx].copy()
    best_length = float(lengths[best_idx])

    stagnant = 0

    for _ in range(generations):
        new_population = np.empty_like(population)

        elite_indices = np.argsort(lengths)[:elite_count]
        new_population[:elite_count] = population[elite_indices]

        write = elite_count

        while write < pop_size:
            parent1 = tournament_select(population, lengths, tournament_size)
            parent2 = tournament_select(population, lengths, tournament_size)

            child1 = order_crossover(parent1, parent2)
            swap_mutation(child1, mutation_rate)
            new_population[write] = child1
            write += 1

            if write < pop_size:
                child2 = order_crossover(parent2, parent1)
                swap_mutation(child2, mutation_rate)
                new_population[write] = child2
                write += 1

        population = new_population
        lengths = population_lengths(population)

        gen_best_idx = int(np.argmin(lengths))
        gen_best_length = float(lengths[gen_best_idx])

        if gen_best_length < best_length:
            best_length = gen_best_length
            best_tour = population[gen_best_idx].copy()
            stagnant = 0
        else:
            stagnant += 1

        if stagnant > 100:
            mutation_rate = min(0.5, mutation_rate * 1.05)
        else:
            mutation_rate = max(0.10, mutation_rate * 0.995)

    return best_tour.tolist(), float(tour_length(best_tour))
