import numpy as np


def _tour_lengths(population, dist):
    """Vectorized closed-tour length computation for a population."""
    return dist[population, np.roll(population, -1, axis=1)].sum(axis=1)


def _tour_length(tour, dist):
    """Closed-tour length for one permutation."""
    tour = np.asarray(tour, dtype=int)
    return float(dist[tour, np.roll(tour, -1)].sum())


def _tournament_select(fitness, rng, k=3):
    """Return the index of the best individual among k random candidates."""
    candidates = rng.integers(0, len(fitness), size=k)
    return candidates[np.argmin(fitness[candidates])]


def _order_crossover(parent1, parent2, rng):
    """
    Order Crossover, OX.

    A slice from parent1 is copied into the child; remaining positions are filled
    in the relative order they appear in parent2.
    """
    n = len(parent1)
    child = np.full(n, -1, dtype=int)

    a, b = rng.integers(0, n, size=2)
    if a > b:
        a, b = b, a

    child[a:b + 1] = parent1[a:b + 1]

    used = np.zeros(n, dtype=bool)
    used[child[a:b + 1]] = True

    fill_pos = (b + 1) % n
    for offset in range(n):
        gene = parent2[(b + 1 + offset) % n]
        if not used[gene]:
            child[fill_pos] = gene
            fill_pos = (fill_pos + 1) % n

    return child


def _swap_mutation(individual, rng):
    """Swap two cities in the permutation."""
    n = len(individual)
    if n < 2:
        return

    i = rng.integers(0, n)
    j = rng.integers(0, n - 1)
    if j >= i:
        j += 1

    individual[i], individual[j] = individual[j], individual[i]


def solve(dist_matrix):
    dist = np.asarray(dist_matrix, dtype=float)
    n = dist.shape[0]

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0

    rng = np.random.default_rng()

    # Adaptive GA parameters.
    if n <= 50:
        pop_size = max(80, 6 * n)
        generations = 600
    elif n <= 200:
        pop_size = 250
        generations = 400
    elif n <= 1000:
        pop_size = 200
        generations = 250
    else:
        pop_size = 100
        generations = 150

    elite_count = max(1, pop_size // 20)
    tournament_k = 3
    crossover_rate = 0.90
    mutation_rate = 0.20

    # Initial population: random permutations.
    population = np.empty((pop_size, n), dtype=int)
    for i in range(pop_size):
        population[i] = rng.permutation(n)

    fitness = _tour_lengths(population, dist)

    best_idx = int(np.argmin(fitness))
    best_tour = population[best_idx].copy()
    best_length = float(fitness[best_idx])

    stagnation = 0
    stagnation_limit = max(60, generations // 4)

    for _ in range(generations):
        new_population = np.empty_like(population)

        # Elitism: copy the best individuals unchanged.
        elite_indices = np.argpartition(fitness, elite_count - 1)[:elite_count]
        new_population[:elite_count] = population[elite_indices]

        pos = elite_count

        while pos < pop_size:
            p1_idx = _tournament_select(fitness, rng, tournament_k)
            p2_idx = _tournament_select(fitness, rng, tournament_k)

            parent1 = population[p1_idx]
            parent2 = population[p2_idx]

            if rng.random() < crossover_rate:
                child1 = _order_crossover(parent1, parent2, rng)
                child2 = _order_crossover(parent2, parent1, rng)
            else:
                child1 = parent1.copy()
                child2 = parent2.copy()

            if rng.random() < mutation_rate:
                _swap_mutation(child1, rng)
            if rng.random() < mutation_rate:
                _swap_mutation(child2, rng)

            new_population[pos] = child1
            pos += 1

            if pos < pop_size:
                new_population[pos] = child2
                pos += 1

        population = new_population
        fitness = _tour_lengths(population, dist)

        gen_best_idx = int(np.argmin(fitness))
        gen_best_length = float(fitness[gen_best_idx])

        if gen_best_length < best_length:
            best_length = gen_best_length
            best_tour = population[gen_best_idx].copy()
            stagnation = 0
        else:
            stagnation += 1

        if stagnation >= stagnation_limit:
            break

    tour = best_tour.tolist()
    length = _tour_length(best_tour, dist)

    return tour, length
