import math
import numpy as np


def solve(dist_matrix):
    D = np.asarray(dist_matrix, dtype=float)
    n = D.shape[0]

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0
    if n == 2:
        return [0, 1], float(D[0, 1] + D[1, 0])

    rng = np.random.default_rng(12345)

    def tour_length(tour):
        return float(D[tour, np.roll(tour, -1)].sum())

    def rotate_to_zero(tour):
        pos = int(np.where(tour == 0)[0][0])
        return np.concatenate((tour[pos:], tour[:pos]))

    def nearest_neighbor(start):
        tour = np.empty(n, dtype=np.int64)
        unvisited = np.ones(n, dtype=bool)
        current = int(start)

        for k in range(n):
            tour[k] = current
            unvisited[current] = False
            if k == n - 1:
                break
            masked = np.where(unvisited, D[current], np.inf)
            current = int(np.argmin(masked))

        return tour

    def two_opt_delta(tour, i, j):
        a = tour[i - 1]
        b = tour[i]
        c = tour[j]
        d = tour[(j + 1) % n]
        return float(D[a, c] + D[b, d] - D[a, b] - D[c, d])

    def random_two_opt_indices():
        # Choose two distinct positions from 1..n-1, keeping city 0 fixed at tour[0].
        m = n - 1
        p = int(rng.integers(0, m))
        q = int(rng.integers(0, m - 1))
        if q >= p:
            q += 1
        i, j = p + 1, q + 1
        if i > j:
            i, j = j, i
        return i, j

    best_initial_len = math.inf

    for s in starts:
        candidate = rotate_to_zero(nearest_neighbor(s))
        candidate_len = tour_length(candidate)
        if candidate_len < best_initial_len:
            best_initial = candidate
            best_initial_len = candidate_len

    current_tour = best_initial.copy()
    current_length = best_initial_len
    best_tour = current_tour.copy()
    best_length = current_length

    # Estimate an initial temperature from random uphill 2-opt moves.
    positive_deltas = []
    samples = min(1000, max(100, 10 * n))
    for _ in range(samples):
        i, j = random_two_opt_indices()
        delta = two_opt_delta(current_tour, i, j)
        if delta > 1e-12:
            positive_deltas.append(delta)

    if positive_deltas:
        initial_temp = float(np.mean(positive_deltas)) / (-math.log(0.8))
    else:
        nonzero = D[D > 0]
        initial_temp = float(np.mean(nonzero)) if nonzero.size else 1.0

    if not math.isfinite(initial_temp) or initial_temp <= 0.0:
        initial_temp = 1.0

    # Simulated annealing with a geometric cooling schedule.
    max_iter = int(max(10_000, min(750_000, 2500 * n)))
    final_temp_fraction = 1e-4
    alpha = final_temp_fraction ** (1.0 / max_iter)
    temp = initial_temp

    for _ in range(max_iter):
        i, j = random_two_opt_indices()
        delta = two_opt_delta(current_tour, i, j)

        if delta <= 0.0 or rng.random() < math.exp(-delta / temp):
            current_tour[i:j + 1] = current_tour[i:j + 1][::-1]
            current_length += delta

            if current_length < best_length:
                best_length = current_length
                best_tour = current_tour.copy()

        temp *= alpha

    # Final deterministic first-improvement 2-opt cleanup for moderate n.
    if n <= 250:

        passes = 0
