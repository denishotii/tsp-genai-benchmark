## Reasoning

**Problem:** Find a near-optimal closed tour visiting all n cities exactly once, minimizing total distance.

**Algorithm components:**
- **Initial solution:** Nearest-neighbor heuristic from city 0 for a good starting point.
- **Neighborhood (2-opt):** Pick indices i<j, reverse tour[i..j]. The change in cost can be computed in O(1) using the four affected edges.
- **Acceptance:** Always accept improving moves; accept worse with probability exp(-Δ/T).
- **Cooling:** Geometric T ← T·α.
- **Initial temperature:** Calibrated from average edge cost so initial acceptance prob ~0.5 for moderate worsening.

**Data structures:** tour as a numpy int array; distance matrix accessed directly.

**Main loop:** For each temperature, run a number of inner iterations. Sample i,j uniformly, compute Δ via four edges (handle wrap-around), accept by Metropolis. Track best tour.

**Edge cases:**
- n ≤ 1: return trivial.
- n == 2: only one tour.
- n == 3: 2-opt reversals don't change cost (symmetric), just return NN tour.
- Avoid i=0, j=n-1 simultaneously (that reverses the whole tour, equivalent move).

**2-opt delta:** For reversing tour[i..j] (with i>0 and j<n-1 ideally, or handling wrap): edges removed are (tour[i-1], tour[i]) and (tour[j], tour[j+1 mod n]); added are (tour[i-1], tour[j]) and (tour[i], tour[j+1 mod n]).

```python
import numpy as np
import random
import math


def _tour_length(tour, D):
    return float(D[tour, np.roll(tour, -1)].sum())


def _nearest_neighbor(D, start=0):
    n = D.shape[0]
    visited = [False] * n
    tour = [start]
    visited[start] = True
    cur = start
    for _ in range(n - 1):
        best, best_d = -1, np.inf
        row = D[cur]
        for j in range(n):
            if not visited[j] and row[j] < best_d:
                best_d = row[j]
                best = j
        tour.append(best)
        visited[best] = True
        cur = best
    return tour


def solve(dist_matrix):
    D = np.asarray(dist_matrix, dtype=float)
    n = D.shape[0]

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0
    if n == 2:
        return [0, 1], float(D[0, 1] + D[1, 0])

    # Initial solution: nearest neighbor
    tour = np.array(_nearest_neighbor(D, 0), dtype=np.int64)
    cur_len = _tour_length(tour, D)

    best_tour = tour.copy()
    best_len = cur_len

    if n == 3:
        return best_tour.tolist(), float(best_len)

    # Calibrate initial temperature from sample of random 2-opt deltas
    rng = random.Random(12345)
    np_rng = np.random.default_rng(12345)

    sample_deltas = []
    for _ in range(min(200, n * n)):
        i = rng.randint(1, n - 2)
        j = rng.randint(i, n - 2)
        if i == j:
            continue
        a, b = tour[i - 1], tour[i]
        c, d = tour[j], tour[j + 1]
        delta = (D[a, c] + D[b, d]) - (D[a, b] + D[c, d])
        if delta > 0:
            sample_deltas.append(delta)

    if sample_deltas:
        avg_pos_delta = sum(sample_deltas) / len(sample_deltas)
        T0 = avg_pos_delta / math.log(2.0)  # ~50% acceptance for typical worsening
    else:
        T0 = 1.0

    T_min = T0 * 1e-6 if T0 > 0 else 1e-10
    alpha = 0.9995

    # Number of iterations: scale with problem size
    inner_iters = max(100, n * 20)
    max_outer = 2000
    # Cap total work
    total_budget = max(50_000, min(500_000, n * n * 50))
    outer = min(max_outer, max(1, total_budget // inner_iters))

    T = T0 if T0 > 0 else 1.0

    for _ in range(int(outer)):
        if T < T_min:
            break
        for _ in range(inner_iters):
            # Pick i, j with 1 <= i < j <= n-1, but avoid (i=1, j=n-1) reversing all
            i = rng.randint(1, n - 2)
            j = rng.randint(i + 1, n - 1)
            if i == 1 and j == n - 1:
                # this reverses interior; equivalent reversal — still valid, allow it
                pass

            a = tour[i - 1]
            b = tour[i]
            c = tour[j]
            # next index wraps
            d = tour[(j + 1) % n]

            # If j+1 wraps to i-1 (only if i-1 == 0 and j == n-1, then d = tour[0] = tour[i-1] when i=1)
            # In that case reversing whole interior is symmetric; delta should be 0 by symmetry.
            removed = D[a, b] + D[c, d]
            added = D[a, c] + D[b, d]
            delta = added - removed

            if delta < 0 or (T > 0 and rng.random() < math.exp(-delta / T)):
                # Apply 2-opt: reverse tour[i..j]
                tour[i:j + 1] = tour[i:j + 1][::-1]
                cur_len += delta
                if cur_len < best_len:
                    best_len = cur_len
                    best_tour = tour.copy()
        T *= alpha

    # Final cleanup: a quick full 2-opt pass for polish
    improved = True
    passes = 0
    while improved and passes < 4:
        improved = False
        passes += 1
        for i in range(1, n - 1):
            a = best_tour[i - 1]
            b = best_tour[i]
            for j in range(i + 1, n):
                c = best_tour[j]
                d = best_tour[(j + 1) % n]
                if a == d:
                    continue
                delta = (D[a, c] + D[b, d]) - (D[a, b] + D[c, d])
                if delta < -1e-12:
                    best_tour[i:j + 1] = best_tour[i:j + 1][::-1]
                    best_len += delta
                    improved = True
                    b = best_tour[i]
        # recompute to avoid drift
        best_len = _tour_length(best_tour, D)

    return best_tour.tolist(), float(best_len)
```
