import numpy as np
import random
import math


def tour_length(tour, dist_matrix):
    idx = np.asarray(tour)
    return float(dist_matrix[idx, np.roll(idx, -1)].sum())


def nearest_neighbor_tour(dist_matrix, start=0):
    n = dist_matrix.shape[0]
    visited = [False] * n
    tour = [start]
    visited[start] = True
    current = start
    for _ in range(n - 1):
        d = dist_matrix[current].copy()
        d[visited] = np.inf
        nxt = int(np.argmin(d))
        tour.append(nxt)
        visited[nxt] = True
        current = nxt
    return tour


def two_opt_delta(tour, i, j, dist_matrix):
    # Reverse tour[i..j]. Edges removed: (i-1,i) and (j,j+1).
    # Edges added: (i-1,j) and (i,j+1).
    n = len(tour)
    a = tour[i - 1]
    b = tour[i]
    c = tour[j]
    d = tour[(j + 1) % n]
    return (dist_matrix[a, c] + dist_matrix[b, d]
            - dist_matrix[a, b] - dist_matrix[c, d])


def solve(dist_matrix):
    dist_matrix = np.asarray(dist_matrix, dtype=float)
    n = dist_matrix.shape[0]

    if n <= 1:
        return list(range(n)), 0.0
    if n == 2:
        return [0, 1], float(dist_matrix[0, 1] + dist_matrix[1, 0])

    # Initial solution via nearest neighbor
    tour = nearest_neighbor_tour(dist_matrix, start=0)
    current_len = tour_length(tour, dist_matrix)
    best_tour = tour[:]
    best_len = current_len

    # Initial temperature: estimate from random neighbor deltas
    sample_deltas = []
    for _ in range(min(100, n * (n - 1) // 2)):
        i = random.randint(1, n - 2)
        j = random.randint(i, n - 1)
        if i == j:
            continue
        d = two_opt_delta(tour, i, j, dist_matrix)
        if d > 0:
            sample_deltas.append(d)
    if sample_deltas:
        avg_pos = sum(sample_deltas) / len(sample_deltas)
        T0 = avg_pos / math.log(2.0)  # ~50% acceptance for avg positive move
    else:
        T0 = 1.0
    T = max(T0, 1e-9)
    T_min = T0 * 1e-6 if T0 > 0 else 1e-9

    # Cooling parameters
    iters_per_temp = max(100, n * n)
    alpha = 0.995
    max_total_iters = max(20000, 200 * n * n)
    max_total_iters = min(max_total_iters, 2_000_000)

    total = 0
    no_improve_steps = 0
    no_improve_limit = max(10000, 50 * n * n)
    no_improve_limit = min(no_improve_limit, 500_000)

    while T > T_min and total < max_total_iters:
        improved_this_temp = False
        for _ in range(iters_per_temp):
            total += 1
            i = random.randint(1, n - 1)
            j = random.randint(1, n - 1)
            if i == j:
                continue
            if i > j:
                i, j = j, i
            if i == 1 and j == n - 1:
                continue  # full reversal does nothing for closed tour
            delta = two_opt_delta(tour, i, j, dist_matrix)
            if delta < 0 or random.random() < math.exp(-delta / T):
                tour[i:j + 1] = tour[i:j + 1][::-1]
                current_len += delta
                if current_len < best_len - 1e-12:
                    best_len = current_len
                    best_tour = tour[:]
                    improved_this_temp = True
                    no_improve_steps = 0
            if total >= max_total_iters:
                break
        if not improved_this_temp:
            no_improve_steps += iters_per_temp
            if no_improve_steps >= no_improve_limit:
                break
        T *= alpha

    # Final 2-opt local refinement
    improved = True
    while improved:
        improved = False
        for i in range(1, n - 1):
            for j in range(i + 1, n):
                if i == 1 and j == n - 1:
                    continue
                delta = two_opt_delta(best_tour, i, j, dist_matrix)
                if delta < -1e-12:
                    best_tour[i:j + 1] = best_tour[i:j + 1][::-1]
                    best_len += delta
                    improved = True
        if n > 200:
            break  # avoid heavy refinement on large instances

    best_len = tour_length(best_tour, dist_matrix)
    return best_tour, float(best_len)
