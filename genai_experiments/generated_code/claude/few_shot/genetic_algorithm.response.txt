import numpy as np
import random


def solve(dist_matrix):
    dist_matrix = np.asarray(dist_matrix, dtype=float)
    n = dist_matrix.shape[0]

    if n == 0:
        return [], 0.0
    if n == 1:
        return [0], 0.0
    if n == 2:
        return [0, 1], float(dist_matrix[0, 1] + dist_matrix[1, 0])

    # GA parameters
    pop_size = max(50, min(200, 10 * n))
    generations = max(100, min(500, 20 * n))
    tournament_size = 5
    mutation_rate = 0.2
    elitism = max(1, pop_size // 20)
    crossover_rate = 0.9

    def tour_length(tour):
        idx = np.asarray(tour)
        return float(dist_matrix[idx, np.roll(idx, -1)].sum())

    def random_tour():
        t = list(range(n))
        random.shuffle(t)
        return t

    def nearest_neighbor_tour(start):
        unvisited = set(range(n))
        unvisited.remove(start)
        tour = [start]
        current = start
        while unvisited:
            nxt = min(unvisited, key=lambda c: dist_matrix[current, c])
            tour.append(nxt)
            unvisited.remove(nxt)
            current = nxt
        return tour

    # Initialize population: mix of NN tours and random tours
    population = []
    num_nn = min(n, pop_size // 5)
    for i in range(num_nn):
        population.append(nearest_neighbor_tour(i % n))
    while len(population) < pop_size:
        population.append(random_tour())

    fitness = [tour_length(t) for t in population]

    def tournament_select():
        best_idx = random.randrange(pop_size)
        best_fit = fitness[best_idx]
        for _ in range(tournament_size - 1):
            i = random.randrange(pop_size)
            if fitness[i] < best_fit:
                best_idx = i
                best_fit = fitness[i]
        return population[best_idx]

    def order_crossover(p1, p2):
        a, b = sorted(random.sample(range(n), 2))
        child = [-1] * n
        child[a:b+1] = p1[a:b+1]
        used = set(p1[a:b+1])
        fill_pos = (b + 1) % n
        p2_pos = (b + 1) % n
        while -1 in child:
            gene = p2[p2_pos]
            if gene not in used:
                child[fill_pos] = gene
                used.add(gene)
                fill_pos = (fill_pos + 1) % n
            p2_pos = (p2_pos + 1) % n
        return child

    def swap_mutate(tour):
        t = tour[:]
        i, j = random.sample(range(n), 2)
        t[i], t[j] = t[j], t[i]
        return t

    best_idx = int(np.argmin(fitness))
    best_tour = population[best_idx][:]
    best_len = fitness[best_idx]

    for gen in range(generations):
        # Sort by fitness for elitism
        order = sorted(range(pop_size), key=lambda i: fitness[i])
        new_pop = [population[order[i]][:] for i in range(elitism)]
        new_fit = [fitness[order[i]] for i in range(elitism)]

        while len(new_pop) < pop_size:
            p1 = tournament_select()
            p2 = tournament_select()
            if random.random() < crossover_rate:
                child = order_crossover(p1, p2)
            else:
                child = p1[:]
            if random.random() < mutation_rate:
                child = swap_mutate(child)
            new_pop.append(child)
            new_fit.append(tour_length(child))

        population = new_pop
        fitness = new_fit

        cur_best = int(np.argmin(fitness))
        if fitness[cur_best] < best_len:
            best_len = fitness[cur_best]
            best_tour = population[cur_best][:]

    return list(best_tour), float(best_len)
